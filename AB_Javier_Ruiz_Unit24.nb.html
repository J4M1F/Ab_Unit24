Luchadores_data <- bind_rows(
  # Funcin para unificar datos de luchadores
  ufc_master %>%
    select(
      #Datos Esquina Roja
      Nombre = RedFighter,
      Altura_Cms = RedHeightCms,
      Alcanze_Cms = RedReachCms,
      Peso_Lbs = RedWeightLbs,
      wins_ko = RedWinsByKO,
      wins_sub = RedWinsBySubmission,
      wins_dec = RedWinsByDecisionUnanimous,
      total_wins = RedWins,
      total_losses = RedLosses,
      stance = RedStance,
      date = Date,
      sig_strikes = RedAvgSigStrLanded,
      td_landed = RedAvgTDLanded,
      sub_att = RedAvgSubAtt,
      td_pct = RedAvgTDPct
    ) %>% 
    mutate(corner = "Rojo"),
ufc_master%>%
  select(
    #Datos Esquina Azul
      Nombre = BlueFighter,
      Altura_Cms = BlueHeightCms,
      Alcanze_Cms = BlueReachCms,
      Peso_Lbs = BlueWeightLbs,
      wins_ko = BlueWinsByKO,
      wins_sub = BlueWinsBySubmission,
      wins_dec = RedWinsByDecisionUnanimous,
      total_wins = BlueWins,
      total_losses = BlueLosses,
      stance = BlueStance,
      date = Date,
      sig_strikes = BlueAvgSigStrLanded,
      td_landed = BlueAvgTDLanded,
      sub_att = BlueAvgSubAtt,
      td_pct = BlueAvgTDPct
    ) %>% 
    mutate(corner = "Azul")
) %>%
  #Agrupamos por luchador para obtener las stats ms recientes
  group_by(Nombre)%>%
  arrange(desc(date))%>%
  slice(1)%>%
  ungroup()%>%
  #Mtricas adicionales
  mutate(
    peleas_totales = total_wins + total_losses,
    rato_victoria = total_wins / peleas_totales,
    
    ratio_ko = ifelse(total_wins >0, wins_ko / total_wins, 0),
    ratio_sum = ifelse(total_wins>0, wins_sub / total_wins, 0),
    ratio_dec = ifelse(total_wins>0, wins_dec / total_wins, 0)
  )%>%
  #Eliminamos valores NA y problematicos
  filter(
    !is.na(Altura_Cms),
    !is.na(Alcanze_Cms),
    !is.na(Peso_Lbs),
    peleas_totales > 0
  )

Grupo 1 (Rojo):

Cualidades: Actividad reducida tanto en impactos importantes como en derribos. Interpretacin: Este conjunto podra simbolizar a los peleadores defensivos o a aquellos que aspiran a dominar las peleas sin tener en cuenta el volumen de impactos o de levantamientos.

Agrupacin 2 (Verde):

Cualidades: Alta actividad en impactos notables cada minuto, aunque con escasos o nulos derribos. Interpretacin: Es probable que sean peleadores de pie expertos en striking, que se concentran en mantener el combate lejos del suelo.

Grupo 3 (Cian):

Cualidades: Minima frecuencia de impactos importantes, pero elevada tasa de derribos por ronda. Interpretacin: Este conjunto parece reunir luchadores que se apoyan en tcnicas de grappling y control sobre el suelo, tales como luchadores o expertos en sumisiones.

Agrupacin 4 (Morado):

Cualidades: Un grupo intermedio que presenta una actividad moderada en ambas medidas. Interpretacin: Simboliza un balance entre el striking y el grappling, sealando a peleadores adaptables que combinan estos estilos.
